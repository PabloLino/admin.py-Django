from django.contrib import admin
from dbmanager import models
from dbmanager import views
from django.contrib.admin import FieldListFilter

# Register your models here.

@admin.register(models.MeusBancos)
class MeusBancos(admin.ModelAdmin):
    def get_queryset(self, request):
        qs = super().get_queryset(request)
        # Superusuários podem ver todos os registros
        if request.user.is_superuser:
            return qs
        # Grupos "backup", "Teste" e "Desenvolvimento" podem ver todos os registros
        if request.user.groups.filter(name__in=['backup', 'Teste', 'Desenvolvimento']).exists():
            return qs
        # Usuários comuns veem apenas seus próprios registros
        return qs.filter(owner=request.user)

    def has_delete_permission(self, request, obj=None):
        # Permitir que o grupo "backup" delete qualquer registro
        if request.user.groups.filter(name='backup').exists():
            return True
        # Permitir que "Teste" e "Desenvolvimento" deletem apenas seus próprios registros
        if request.user.groups.filter(name__in=['Teste', 'Desenvolvimento']).exists():
            if obj is not None and obj.owner == request.user:
                return True
        # Caso contrário, negar permissão
        return False

    def get_fieldsets(self, request, obj=None):
        # Superusuários e grupo "backup" podem editar todos os campos
        if request.user.groups.filter(name='backup').exists():
            return [
                (None, {'fields': ('owner', 'servidor', 'nmbanco', 'estouusando')}),
            ]
        # Usuários comuns editam apenas campos básicos
        return [
            (None, {'fields': ('servidor', 'nmbanco', 'estouusando')}),
        ]

    list_display = 'owner', 'servidor', 'nmbanco', 'estouusando', 'dtcriacao',
    list_filter = ('owner', 'estouusando', 'dtcriacao', 'servidor',)
    list_editable = 'estouusando',
    search_fields = ('nmbanco',)
    ordering = 'dtcriacao',

    def save_model(self, request, obj, form, change):
        # Garantir que apenas o grupo "backup" pode alterar o proprietário
        if not request.user.groups.filter(name='backup').exists():
            obj.owner = request.user
        super().save_model(request, obj, form, change)
